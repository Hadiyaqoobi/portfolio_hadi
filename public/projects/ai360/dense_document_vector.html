def GetWV(wv, sWord='nlp') -> np.array:
    ''' Returns a word vector for sWord (in lower case), if it is found, 
        and a zero vector (of length n) otherwise, where n is the length of vectors in wv.
    wv: Gensim's word2vec model object'''
    
    s = str(sWord).lower()
    if s in wv:
        return wv.get_vector(s)
    return np.zeros(wv.vector_size)
def GetSupWords(wv, sWord='nlp') -> [str]:
    '''Return all wv vocabulary words, for which sWord (in lower case) 
        is a subword, i.e. a substring. If none is found, return an empty list.
    wv: Gensim's word2vec model '''
    
    sub = str(sWord).lower()
    out = []
    
    for w in wv.index_to_key:  # preserves model order (critical for tests)
        if sub in str(w).lower():
            out.append(w)
    return out
def NN(wv, sWord='pet', nThreshold=0.75) -> [(str, float)]:
    '''For sWord (in lower-case), return a list of most similar words 
    and corresponding similarity score. Only similarities above nThreshold are returned.
    If none is found, return an empty list.'''
    
    s = str(sWord).lower()
    if s not in wv:
        return []
    
    # CRITICAL: Use topn=100 to get enough neighbors
    sims = wv.most_similar(s, topn=100)
    return [(w, sim) for (w, sim) in sims if sim >= nThreshold]
def NN2(wv, LsWords) -> tuple:
    '''Return a tuple (similarity, word1, word2) for the pair of words with the highest similarity.
    LsWords: list of words
    Return: (similarity, word1, word2) where word1 < word2 alphabetically
            If fewer than 2 valid words, return None'''
    
    # Convert all words to lowercase and filter to those in vocabulary
    valid_words = []
    for word in LsWords:
        w_lower = str(word).lower()
        if w_lower in wv:
            valid_words.append(w_lower)
    
    # Need at least 2 words to find a pair
    if len(valid_words) < 2:
        return None
    
    # Find the pair with maximum similarity
    max_sim = -1
    best_pair = None
    
    for i in range(len(valid_words)):
        for j in range(i + 1, len(valid_words)):
            w1, w2 = valid_words[i], valid_words[j]
            sim = wv.similarity(w1, w2)
            
            if sim > max_sim:
                max_sim = sim
                # Return alphabetically ordered pair
                if w1 < w2:
                    best_pair = (sim, w1, w2)
                else:
                    best_pair = (sim, w2, w1)
    
    return best_pair
def NNExc(wv, sWord='pet', LsExcept=['cat','dog']) -> (str, float):
    ''' Lower-case all input words and use Gensim's most_similar() 
        to find sWord's neighbor, which is not in LsExcept list.
    wv: Gensim's word2vec model  
    Return: a tuple with (neighbor X, similarity score between X and sWord)
        If none is found, return None.'''
    
    s = str(sWord).lower()
    
    # Check if word is in vocabulary
    if s not in wv:
        return None
    
    # Convert exception list to lowercase
    except_lower = [str(w).lower() for w in LsExcept]
    
    # Get most similar words (use large topn to ensure we find one not in exception list)
    similar_words = wv.most_similar(s, topn=100)
    
    # Find first word not in exception list
    for word, similarity in similar_words:
        if word not in except_lower:
            return (word, similarity)
    
    # If all similar words are in exception list
    return None
def NNChain(wv, sWord='pet', n=5) -> [(str, float)]:
    ''' For the lower-cased sWord find a chain of n words where each word is the closest
        neighbor of the previous word excluding all words chained so far, including sWord.
        Use NNExc() to find the next neighbor given words in a chain + sWord as the exclusion list.
    Example: 'cat' neighbors with 'dog' with similarity .92; 
             'dog' neighbors with 'dogs' ('cat' was already used), and so on.
    Return a list of chained words with their corresponding similarity scores 
            (between the word and its previous neighbor).
        If none is found, return an empty list. '''
    
    s = str(sWord).lower()
    
    # Check if starting word is in vocabulary
    if s not in wv:
        return []
    
    chain = []
    exclusion_list = [s]  # Start by excluding the original word
    current_word = s
    
    for _ in range(n):
        # Find next neighbor excluding all words used so far
        result = NNExc(wv, current_word, exclusion_list)
        
        if result is None:
            # No more neighbors found
            break
        
        next_word, similarity = result
        chain.append((next_word, similarity))
        exclusion_list.append(next_word)
        current_word = next_word
    
    return chain
