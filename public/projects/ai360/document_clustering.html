def MakeMask(df=df, tok={'pet','pets','animal','animals','cat','dog','dogs'}) -> [bool]:
    if 'overview' not in df.columns:
        return []
    LbMask = []
    for overview in df['overview']:
        words = set(nltk.word_tokenize(overview.lower()))
        LbMask.append(bool(words & tok))
    return LbMask

def sil1(dfEmb=dfEmb, K=3, ReturnBest=False) -> pd.DataFrame():
    dfOut = pd.DataFrame([], columns=['SIL', 'k'])
    N = len(dfEmb)
    K = min(K, N - 1)  # reset K if larger than N-1
    rows = []
    for k in range(2, K + 1):
        labels = hc(dfEmb, k=k)
        score = SIL(dfEmb, labels)
        rows.append({'SIL': score, 'k': k})
    dfOut = pd.DataFrame(rows, columns=['SIL', 'k'])
    if ReturnBest:
        dfOut = dfOut[dfOut['SIL'] == dfOut['SIL'].max()]
    return dfOut

def sil2(dfEmb, K=2, linkages=['ward'], ReturnBest=False) -> pd.DataFrame():
    if not isinstance(linkages, list): linkages = [linkages]
    dfOut = pd.DataFrame([], columns=['SIL', 'k', 'link'])
    N = len(dfEmb)
    K = min(K, N - 1)
    rows = []
    for linkage in linkages:
        for k in range(2, K + 1):
            labels = hc(dfEmb, k=k, linkage=linkage)
            score = SIL(dfEmb, labels)
            rows.append({'SIL': score, 'k': k, 'link': linkage})
    dfOut = pd.DataFrame(rows, columns=['SIL', 'k', 'link'])
    if ReturnBest:
        dfOut = dfOut[dfOut['SIL'] == dfOut['SIL'].max()]
    return dfOut
