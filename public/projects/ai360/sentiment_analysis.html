def ApplySIA(LsPos=[''], LsNeg=[''], sia=sia) -> pd.DataFrame:
    rows = []
    for review in LsPos:
        scores = sia.polarity_scores(review)
        scores['vY'] = 1
        scores['pY'] = -1 if scores['compound'] < 0 else 1
        scores['review'] = review
        rows.append(scores)
    for review in LsNeg:
        scores = sia.polarity_scores(review)
        scores['vY'] = -1
        scores['pY'] = -1 if scores['compound'] < 0 else 1
        scores['review'] = review
        rows.append(scores)
    df = pd.DataFrame(rows, columns=['neg', 'neu', 'pos', 'compound', 'vY', 'pY', 'review'])
    return df

def OOV(Docs=df1.review, Vocab=sia.lexicon) -> set(''):
    all_words = ' '.join(Docs).lower().split()
    SsNewWords = {w for w in all_words
                  if len(w) > 1
                  and w.isalpha()
                  and w not in Vocab}
    return SsNewWords

def Expand(SsWords={''}) -> list(''):
    SsWordsNew = set(SsWords)
    for w in SsWords:
        for pos in [wn.NOUN, wn.ADJ, wn.VERB]:
            morphed = wn.morphy(w, pos)
            if morphed:
                SsWordsNew.add(morphed)
    return sorted(SsWordsNew)
