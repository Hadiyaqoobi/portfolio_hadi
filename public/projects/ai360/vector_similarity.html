def LCV(wv, LsWords=['NLP','is','fun','!']) -> [str]:
    out = []
    for word in LsWords:
        w_lower = str(word).lower()
        if w_lower in wv:
            out.append(w_lower)
    return out
def DV(wv, LsWords=['I','like','nlp']) -> np.array([]):
    if isinstance(LsWords, str):
        LsWords = nltk.word_tokenize(LsWords)
    lcv_words = LCV(wv, LsWords)
    if len(lcv_words) == 0:
        return np.zeros(wv.vector_size)
    vectors = [wv[w] for w in lcv_words]
    return np.mean(vectors, axis=0)
def TopWords(wv, LsWords=['super','cow','now'], n=1) -> [(str, float)]:
    lcv_words = LCV(wv, LsWords)
    if len(lcv_words) == 0:
        return None
    unique_words = []
    seen = set()
    for w in lcv_words:
        if w not in seen:
            unique_words.append(w)
            seen.add(w)
    doc_vec = DV(wv, LsWords)
    similarities = []
    for word in unique_words:
        word_vec = wv[word]
        sim = 1 - cosine(word_vec, doc_vec)
        similarities.append((word, sim))
    similarities.sort(key=lambda x: -x[1])
    return similarities[:n]
def NN0(wv, LsWords=['Cat','in','the','hat']) -> (str, float):
    lcv_words = LCV(wv, LsWords)
    if len(lcv_words) == 0:
        return None
    min_norm = float('inf')
    best_word = None
    for word in lcv_words:
        word_vec = wv[word]
        word_norm = norm(word_vec)
        if word_norm < min_norm:
            min_norm = word_norm
            best_word = word
    return (best_word, min_norm)
def NN_Pres(wv, sQryFID='1945-Roosevelt.txt', LvCorpusFID=['1865-Lincoln.txt','1905-Roosevelt.txt']) -> (str, float):
    try:
        query_words = nltk.corpus.inaugural.words(sQryFID)
    except:
        return None
    query_vec = DV(wv, list(query_words))
    if np.all(query_vec == 0):
        return None
    corpus_fids = [fid for fid in LvCorpusFID if fid != sQryFID]
    if len(corpus_fids) == 0:
        return None
    max_sim = -1
    best_fid = None
    for fid in corpus_fids:
        try:
            doc_words = nltk.corpus.inaugural.words(fid)
            doc_vec = DV(wv, list(doc_words))
            if np.all(doc_vec == 0):
                continue
            sim = 1 - cosine(query_vec, doc_vec)
            if sim > max_sim:
                max_sim = sim
                best_fid = fid
        except:
            continue
    if best_fid is None:
        return None
    return (best_fid, max_sim)
def CumMean(x=np.array([0,1,2])) -> np.array([float]):
    if len(x) == 0:
        return np.array([])
    cumsum = np.cumsum(x)
    indices = np.arange(1, len(x) + 1)
    return cumsum / indices
def PickPosValues(wv, LsWords=['NLP','is'], nPos=0) -> [float]:
    lcv_words = LCV(wv, LsWords)
    if len(lcv_words) == 0:
        return []
    out = []
    for word in lcv_words:
        word_vec = wv[word]
        out.append(word_vec[nPos])
    return out
def CumMeanInPos(wv, LsWords=['Hi','from','Mars','!'], nPos=0, nSkip=50) -> np.array([float]):
    pos_values = PickPosValues(wv, LsWords, nPos)
    if len(pos_values) == 0:
        return np.array([])
    cum_mean = CumMean(np.array(pos_values))
    if nSkip >= len(cum_mean):
        return np.array([])
    return cum_mean[nSkip:]
