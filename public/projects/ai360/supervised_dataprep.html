def Split2(X=range(10), j=2):
    X = list(X)
    if not isinstance(j, int) or not (1 <= j <= len(X) - 1):
        j = 2
    tX = [x for i, x in enumerate(X) if i % j != 0]
    vX = [x for i, x in enumerate(X) if i % j == 0]
    return tX, vX
def Split3(X=range(10), i=2, j=3):
    X = list(X)
    if not isinstance(j, int) or not (1 <= j <= len(X) - 1):
        j = 2
    if not isinstance(i, int) or not (1 <= i <= len(X) - 1):
        i = 2
    tvX, sX = Split2(X, j)
    tX, vX = Split2(tvX, i)
    return tX, vX, sX
def KSplit(X=range(10), K=2):
    X = list(X)
    n = len(X)
    if not isinstance(K, int) or not (1 <= K <= n - 1):
        K = 2
    LSamples = [[] for _ in range(K)]
    for i, x in enumerate(X):
        LSamples[i % K].append(x)
    return LSamples
