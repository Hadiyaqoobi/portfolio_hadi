    text = nltk.corpus.inaugural.raw(fid)
    LsSents = nltk.sent_tokenize(text)
    return LsSents

def GetDTM(LsSents=['']):
    tv = TfidfVectorizer()
    sm = tv.fit_transform(LsSents)
    dfDTM = pd.DataFrame(sm.toarray(),
                         index=[s[:50] for s in LsSents],
                         columns=tv.get_feature_names())
    return dfDTM

dfDTM = GetDTM(LsSents)
def GetGramian(dfDTM):
    arr = dfDTM.values
    dfSim = pd.DataFrame(arr @ arr.T, index=dfDTM.index, columns=dfDTM.index)
    return dfSim

dfSim = GetGramian(dfDTM)
def GetGraph(mSim):
    G = nx.from_numpy_array(mSim)
    G.name = 'Sentence Similarities'
    return G

G = GetGraph(dfSim.values)
def RankSents(G, LsSents=['']):
    ranks = nx.pagerank(G)
    dfPgRnk = pd.DataFrame({'Rank': [ranks[i] for i in range(len(LsSents))],
                             'Sent': LsSents})
    dfPgRnk = dfPgRnk.sort_values('Rank', ascending=False)
    return dfPgRnk

dfPgRnk = RankSents(G, LsSents)
