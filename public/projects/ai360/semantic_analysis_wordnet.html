def Lemmas(sLemma='tiger') -> set():
    '''Returns a set of all lemma names from all synsets of sLemma.'''
    SsLemmas = set()
    for ss in wn.synsets(sLemma):
        for name in ss.lemma_names():
            SsLemmas.add(name)
    return SsLemmas

def HypLemmas(sLemma='tiger') -> set():
    '''Returns a set of all lemma names from all the hypernyms of all the synsets of sLemma.'''
    SsLemmas = set()
    for ss in wn.synsets(sLemma):
        for hyp in ss.hypernyms():
            for name in hyp.lemma_names():
                SsLemmas.add(name)
    return SsLemmas

def Sim(sLemma1='teacher', sLemma2='tiger', ReturnBest=False) -> pd.DataFrame():
    '''Computes WordNet path similarity between all pairs of synsets.'''
    rows = []
    for ss1 in wn.synsets(sLemma1):
        for ss2 in wn.synsets(sLemma2):
            sim = wn.path_similarity(ss1, ss2)
            if sim is not None:
                rows.append({'ss1': ss1.name(), 'ss2': ss2.name(), 'sim': sim})
    
    dfSim = pd.DataFrame(rows, columns=['ss1', 'ss2', 'sim'])
    dfSim = dfSim.sort_values(['ss1', 'ss2']).reset_index(drop=True)
    
    if ReturnBest:
        dfSim = dfSim[dfSim['sim'] == dfSim['sim'].max()].reset_index(drop=True)
    
    return dfSim
