def CharVec(w, wv=wv):
    vecs = [wv[c] for c in w if c in wv]
    if len(vecs) == 0:
        return np.zeros(50)
    return np.mean(vecs, axis=0)
def W2V(w, wv=wv):
    if w in wv:
        return wv[w]
    return CharVec(w, wv)
